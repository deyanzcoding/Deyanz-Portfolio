        let currentProject = 0;
        let currentScreen = 0;

        function openModal(event, projectIdx, screenIdx) {
            event.preventDefault();
            currentProject = projectIdx;
            currentScreen = screenIdx;
            updateModalImg();
            document.getElementById('workModal').style.display = 'flex';
        }

        function closeModal() {
            document.getElementById('workModal').style.display = 'none';
            document.getElementById('modalImg').src = '';
        }

        function updateModalImg() {
            const img = document.getElementById('modalImg');
            img.src = projectScreens[currentProject][currentScreen];
            // Hide prev/next if at ends
            document.getElementById('modalPrev').style.visibility = currentScreen > 0 ? 'visible' : 'hidden';
            document.getElementById('modalNext').style.visibility = currentScreen < projectScreens[currentProject].length - 1 ? 'visible' : 'hidden';
        }

        document.getElementById('modalPrev').onclick = function(e) {
            if (currentScreen > 0) {
            currentScreen--;
            updateModalImg();
            }
        };
        document.getElementById('modalNext').onclick = function(e) {
            if (currentScreen < projectScreens[currentProject].length - 1) {
            currentScreen++;
            updateModalImg();
            }
        };
        window.addEventListener('keydown', function(e) {
            if (document.getElementById('workModal').style.display === 'flex') {
            if (e.key === 'ArrowLeft' && currentScreen > 0) {
                currentScreen--;
                updateModalImg();
            }
            if (e.key === 'ArrowRight' && currentScreen < projectScreens[currentProject].length - 1) {
                currentScreen++;
                updateModalImg();
            }
            if (e.key === 'Escape') closeModal();
            }
        });
        document.getElementById('workModal').addEventListener('click', function(e) {
            if (e.target === this) closeModal();
        });
        </script>
